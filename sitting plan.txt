#include <iostream>
using namespace std;

int main() {

    int rows, cols;

    cout << "Enter number of rows: ";
    cin >> rows;

    cout << "Enter number of seats in each row: ";
    cin >> cols;

    int seat[20][20] = {0}; 
    int choice;

    while (true) {

        cout << "\n--- Seat Map ---\n";
        for (int i = 1; i <= rows; i++) {
            for (int j = 1; j <= cols; j++) {

                if (seat[i][j] == 0)
                    cout << "(" << i << "-" << j << " F) ";
                else
                    cout << "(" << i << "-" << j << " R) ";
            }
            cout << endl;
        }

       
        cout << "\nPress 1 to Book a Seat\n";
        cout << "Press 0 to Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        if (choice == 0) {
            cout << "Thank you for using the system!";
            break;
        }

        int r, c;

        cout << "Enter Row Number: ";
        cin >> r;

        cout << "Enter Seat Number: ";
        cin >> c;

       
        if (r < 1 || r > rows || c < 1 || c > cols) {
            cout << "Invalid seat! Try again.\n";
            continue;
        }

      
        if (seat[r][c] == 1) {
            cout << "Seat already reserved!\n";
        }
        else {
            seat[r][c] = 1;
            cout << "Seat booked successfully!\n";
        }
    }

    return 0;
}
