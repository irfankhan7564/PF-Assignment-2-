#include <iostream>
using namespace std;

void reverseArray(int arr[]) {    
    int temp;

    temp = arr[0];
    arr[0] = arr[4];
    arr[4] = temp;
}

int main() {
    int arr[5] = {10, 20, 30, 40, 50};
    cout<<"original array"<<endl;;
    for(int i=0; i<5; i++){
        cout<<arr[i]<<" ";
    }
cout<<endl;

    reverseArray(arr);  

    cout << "Reversed Array";
    cout<<endl;
    for (int i = 0; i < 5; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
